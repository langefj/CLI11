716  std::string get_name(bool positional = false, //<[input] Show the positional name
717  bool all_options = false //<[input] Show every option
718  ) const {
719  if(get_group().empty())
720  return {}; // Hidden
721 
722  if(all_options) {
723 
724  std::vector<std::string> name_list;
725 
727  if((positional && (!pname_.empty())) || (snames_.empty() && lnames_.empty())) {
728  name_list.push_back(pname_);
729  }
730  if((get_items_expected() == 0) && (!fnames_.empty())) {
731  for(const std::string &sname : snames_) {
732  name_list.push_back("-" + sname);
733  if(check_fname(sname)) {
734  name_list.back() += "{" + get_flag_value(sname, "") + "}";
735  }
736  }
737 
738  for(const std::string &lname : lnames_) {
739  name_list.push_back("--" + lname);
740  if(check_fname(lname)) {
741  name_list.back() += "{" + get_flag_value(lname, "") + "}";
742  }
743  }
744  } else {
745  for(const std::string &sname : snames_)
746  name_list.push_back("-" + sname);
747 
748  for(const std::string &lname : lnames_)
749  name_list.push_back("--" + lname);
750  }
751 
752  return detail::join(name_list);
753  }
754 
755  // This returns the positional name no matter what
756  if(positional)
757  return pname_;
758 
759  // Prefer long name
760  if(!lnames_.empty())
761  return std::string(2, '-') + lnames_[0];
762 
763  // Or short name if no long name
764  if(!snames_.empty())
765  return std::string(1, '-') + snames_[0];
766 
767  // If positional is the only name, it's okay to use that
768  return pname_;
769  }

776  void run_callback() {
777 
778  if(current_option_state_ == option_state::parsing) {
779  _validate_results(results_);
780  current_option_state_ = option_state::validated;
781  }
782 
783  if(current_option_state_ < option_state::reduced) {
784  _reduce_results(proc_results_, results_);
785  current_option_state_ = option_state::reduced;
786  }
787  if(current_option_state_ >= option_state::reduced) {
788  current_option_state_ = option_state::callback_run;
789  if(!(callback_)) {
790  return;
791  }
792  const results_t &send_results = proc_results_.empty() ? results_ : proc_results_;
793  bool local_result = callback_(send_results);
794 
795  if(!local_result)
796  throw ConversionError(get_name(), results_);
797  }
798  }

801  const std::string &matching_name(const Option &other) const {
802  static const std::string estring;
803  for(const std::string &sname : snames_)
804  if(other.check_sname(sname))
805  return sname;
806  for(const std::string &lname : lnames_)
807  if(other.check_lname(lname))
808  return lname;
809 
810  if(ignore_case_ ||
811  ignore_underscore_) { // We need to do the inverse, in case we are ignore_case or ignore underscore
812  for(const std::string &sname : other.snames_)
813  if(check_sname(sname))
814  return sname;
815  for(const std::string &lname : other.lnames_)
816  if(check_lname(lname))
817  return lname;
818  }
819  return estring;
820  }

825  bool check_name(std::string name) const {
826 
827  if(name.length() > 2 && name[0] == '-' && name[1] == '-')
828  return check_lname(name.substr(2));
829  if(name.length() > 1 && name.front() == '-')
830  return check_sname(name.substr(1));

836  }
837  if(ignore_case_) {

862  std::string get_flag_value(std::string name, std::string input_value) const {
863  static const std::string trueString{"true"};
864  static const std::string falseString{"false"};
865  static const std::string emptyString{"{}"};
866  // check for disable flag override_
867  if(disable_flag_override_) {
868  if(!((input_value.empty()) || (input_value == emptyString))) {
869  auto default_ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
870  if(default_ind >= 0) {
871  // We can static cast this to size_t because it is more than 0 in this block
872  if(default_flag_values_[static_cast<size_t>(default_ind)].second != input_value) {
873  throw(ArgumentMismatch::FlagOverride(name));
874  }
875  } else {
876  if(input_value != trueString) {
877  throw(ArgumentMismatch::FlagOverride(name));
878  }
879  }
880  }
881  }
882  auto ind = detail::find_member(name, fnames_, ignore_case_, ignore_underscore_);
883  if((input_value.empty()) || (input_value == emptyString)) {
884  if(flag_like_) {
885  return (ind < 0) ? trueString : default_flag_values_[static_cast<size_t>(ind)].second;
886  } else {
887  return (ind < 0) ? default_str_ : default_flag_values_[static_cast<size_t>(ind)].second;
888  }
889  }
890  if(ind < 0) {
891  return input_value;
892  }
893  if(default_flag_values_[static_cast<size_t>(ind)].second == falseString) {
894  try {
895  auto val = detail::to_flag_value(input_value);
896  return (val == 1) ? falseString : (val == (-1) ? trueString : std::to_string(-val));
897  } catch(const std::invalid_argument &) {
898  return input_value;
899  }
900  } else {
901  return input_value;
902  }
903  }

1093  // Run the Validators (can change the string)
1094  if(!validators_.empty()) {
1095  if(type_size_max_ > 1) { // in this context index refers to the index in the type
1096  int index = 0;

1099  // create a negative index for the earliest ones
1100  index = get_items_expected_max() - static_cast<int>(res.size());
1101  }
1102 
1103  for(std::string &result : res) {

1105  index = 0; // reset index for variable size chunks

1109  if(!err_msg.empty())

1112  }
1113  } else {
1114  int index = 0;
1115  if(expected_max_ < static_cast<int>(res.size()) &&
1116  multi_option_policy_ == CLI::MultiOptionPolicy::TakeLast) {
1117  // create a negative index for the earliest ones
1118  index = expected_max_ - static_cast<int>(res.size());
1119  }
1120  for(std::string &result : res) {
1121  auto err_msg = _validate(result, index);
1122  ++index;

1133  void _reduce_results(results_t &res, const results_t &original) const {
1134 
1135  // max num items expected or length of vector, always at least 1
1136  // Only valid for a trimming policy
1137 
1138  res.clear();
1139  // Operation depends on the policy setting
1140  switch(multi_option_policy_) {
1141  case MultiOptionPolicy::TakeAll:
1142  break;
1143  case MultiOptionPolicy::TakeLast:
1144  // Allow multi-option sizes (including 0)
1145  {
1146  size_t trim_size = std::min<size_t>(std::max<size_t>(get_items_expected_max(), 1), original.size());
1147  if(original.size() != trim_size) {
1148  res.assign(original.end() - trim_size, original.end());
1149  }
1150  }
1151  break;
1152  case MultiOptionPolicy::TakeFirst: {
1153  size_t trim_size = std::min<size_t>(std::max<size_t>(get_items_expected_max(), 1), original.size());
1154  if(original.size() != trim_size) {
1155  res.assign(original.begin(), original.begin() + trim_size);
1156  }
1157  } break;
1158  case MultiOptionPolicy::Join:
1159  if(results_.size() > 1) {
1160  res.push_back(detail::join(original, std::string(1, (delimiter_ == '\0') ? '\n' : delimiter_)));
1161  }
1162  break;
1163  case MultiOptionPolicy::Throw:
1164  default: {
1165  auto num_min = static_cast<size_t>(get_items_expected_min());
1166  auto num_max = static_cast<size_t>(get_items_expected_max());
1167  if(num_min == 0) {
1168  num_min = 1;
1169  }
1170  if(num_max == 0) {
1171  num_max = 1;
1172  }
1173  if(original.size() < num_min) {
1174  throw ArgumentMismatch::AtLeast(get_name(), static_cast<int>(num_min), original.size());
1175  }
1176  if(original.size() > num_max) {
1177  throw ArgumentMismatch::AtMost(get_name(), static_cast<int>(num_max), original.size());
1178  }
1179  break;
1180  }
1181  }
1182  }

1184  // Run a result through the Validators
1185  std::string _validate(std::string &result, int index) const {
1186  std::string err_msg;
1187  if(result.empty() && expected_min_ == 0) {
1188  // an empty with nothing expected is allowed
1189  return err_msg;
1190  }
1191  for(const auto &vali : validators_) {
1192  auto v = vali.get_application_index();
1193  if(v == -1 || v == index) {
1194  try {
1195  err_msg = vali(result);
1196  } catch(const ValidationError &err) {
1197  err_msg = err.what();
1198  }
1199  if(!err_msg.empty())
1200  break;
1201  }
1202  }
1203 
1204  return err_msg;
1205  }

1208  int _add_result(std::string &&result, std::vector<std::string> &res) const {
1209  int result_count = 0;
1210  if(allow_extra_args_ && !result.empty() && result.front() == '[' &&
1211  result.back() == ']') { // this is now a vector string likely from the default or user entry
1212  result.pop_back();
1213 
1214  for(auto &var : CLI::detail::split(result.substr(1), ',')) {
1215  if(!var.empty()) {
1216  result_count += _add_result(std::move(var), res);
1217  }
1218  }
1219  return result_count;
1220  }
1221  if(delimiter_ == '\0') {
1222  res.push_back(std::move(result));
1223  ++result_count;
1224  } else {
1225  if((result.find_first_of(delimiter_) != std::string::npos)) {
1226  for(const auto &var : CLI::detail::split(result, delimiter_)) {
1227  if(!var.empty()) {
1228  res.push_back(var);
1229  ++result_count;
1230  }
1231  }
1232  } else {
1233  res.push_back(std::move(result));
1234  ++result_count;
1235  }
1236  }
1237  return result_count;
